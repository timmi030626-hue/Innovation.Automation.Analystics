        title = i.find("h3",class_="_11dVb")

            r1= re.findall("^[0-9]{4}",text)

                brand.append(np.nan)

            if r3:
                model.append(r3[0])
            else:
                model.append(np.nan)          
        
        km_fuel_transmission = i.find("ul",class_="_3J2G-")
        #print(km_fuel_transmission)
        #regular expression required
        if km_fuel_transmission:
            text1 = km_fuel_transmission.text
            
            r4= re.findall("(?:Petrol|Diesel|CNG)",text1)
            if r4:
                fuel_type.append(r4[0])
            else:
                fuel_type.append(np.nan)
              
            r5= re.findall("([0-9]+\,[0-9]+)",text1)
            if r5:
                km_driven.append(r5[0])
            else:
                km_driven.append(np.nan)

            r6= re.findall("[0-9]+\,[0-9]+\skm(?:Petrol|Diesel|CNG)([a-zA-z]+)",text1)
            if r6:
                transmission.append(r6[0])
            else:
                transmission.append(np.nan)            
        price_in_lakhs = i.find("strong",class_="_3RL-I")
        if price_in_lakhs:
            text2 = price_in_lakhs.text
       
            r7= re.findall("([0-9]+\.[0-9]+)",text2)
            if r7:
                price.append(r7[0])
            else:
                price.append(np.nan)        
                
        emi = i.find("span",class_="_2O0yU")
        #print(emi)
        if emi:
            text3 = emi.text
        #regex required   
            r8= re.findall("([0-9]+\,[0-9]+)",text3)
            if r8:
                emi_month.append(r8[0])
            else:
                emi_month.append(np.nan)
